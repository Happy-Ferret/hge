type Flag

type Flag sdl.Mod // HGE Input Event flags (multiple ones may be OR'd)

const (
    INP_SHIFT      Flag = sdl.KMOD_LSHIFT | sdl.KMOD_RSHIFT
    INP_CTRL       Flag = sdl.KMOD_LCTRL | sdl.KMOD_RCTRL
    INP_ALT        Flag = sdl.KMOD_LALT | sdl.KMOD_RALT
    INP_CAPSLOCK   Flag = sdl.KMOD_CAPS
    INP_SCROLLLOCK Flag = sdl.KMOD_MODE
    INP_NUMLOCK    Flag = sdl.KMOD_NUM
    INP_REPEAT     Flag = 0x20000
)

type InputEvent

type InputEvent struct {
    Type   Type    // event type
    Key    Key     // key code
    Flags  Flag    // event flags
    Chr    uint8   // character code
    Button Button  // Mouse Button
    Wheel  int     // wheel shift
    X      float64 // mouse cursor x-coordinate
    Y      float64 // mouse cursor y-coordinate
    // contains filtered or unexported fields
}
